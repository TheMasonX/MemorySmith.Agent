### 2.2 Mapping functions

**YAML → canonical**

```csharp
public static class BlueprintYamlMapper
{
    public static Blueprint ToCanonical(BlueprintYamlDto dto)
    {
        var dimensions = new Dimensions(dto.Dimensions.X, dto.Dimensions.Y, dto.Dimensions.Z);
        var origin = new Origin(dto.Origin.X, dto.Origin.Y, dto.Origin.Z);

        var blocks = dto.Blocks
            .Select(b => new PlacementBlock(
                b.Pos.X,
                b.Pos.Y,
                b.Pos.Z,
                b.Id,
                b.State is null ? null : new BlockState { Properties = b.State },
                (IReadOnlyList<string>)(b.Tags ?? new List<string>())
            ))
            .ToArray();

        var metadata = new BlueprintMetadata
        {
            Category = dto.Metadata.Category,
            Difficulty = dto.Metadata.Difficulty,
            SurvivalReady = dto.Metadata.SurvivalReady,
            Author = dto.Metadata.Author,
            CreatedAt = dto.Metadata.CreatedAt,
            IsTemplate = dto.Metadata.IsTemplate,
            TemplateCategory = dto.Metadata.TemplateCategory
        };

        PhaseConfig? phaseConfig = null;
        if (dto.Phases is not null)
        {
            phaseConfig = new PhaseConfig
            {
                Mode = dto.Phases.Mode.ToLowerInvariant() switch
                {
                    "override" => PhaseMode.Override,
                    "augment"  => PhaseMode.Augment,
                    _          => PhaseMode.Auto
                },
                Phases = dto.Phases.List
                    .Select(p => new BuildPhase
                    {
                        Id = p.Id,
                        Description = p.Description,
                        Selectors = new PhaseSelectors
                        {
                            Tags = (IReadOnlyList<string>)(p.Selectors.Tags ?? new List<string>()),
                            Ranges = (IReadOnlyList<PhaseRange>)(p.Selectors.Ranges ?? new List<PhaseRangeDto>())
                                .Select(r => new PhaseRange { MinY = r.MinY, MaxY = r.MaxY })
                                .ToArray()
                        }
                    })
                    .ToArray()
            };
        }

        return new Blueprint
        {
            Id = dto.Id,
            Name = dto.Name,
            Version = dto.Version,
            Dimensions = dimensions,
            Origin = origin,
            Metadata = metadata,
            Blocks = blocks,
            PhaseConfig = phaseConfig
        };
    }
}
```

**Canonical → YAML**

```csharp
public static BlueprintYamlDto ToYaml(Blueprint blueprint)
{
    var dto = new BlueprintYamlDto
    {
        Id = blueprint.Id,
        Name = blueprint.Name,
        Version = blueprint.Version,
        Dimensions = new DimensionsDto
        {
            X = blueprint.Dimensions.X,
            Y = blueprint.Dimensions.Y,
            Z = blueprint.Dimensions.Z
        },
        Origin = new OriginDto
        {
            X = blueprint.Origin.X,
            Y = blueprint.Origin.Y,
            Z = blueprint.Origin.Z
        },
        Metadata = new BlueprintMetadataDto
        {
            Category = blueprint.Metadata.Category,
            Difficulty = blueprint.Metadata.Difficulty,
            SurvivalReady = blueprint.Metadata.SurvivalReady,
            Author = blueprint.Metadata.Author,
            CreatedAt = blueprint.Metadata.CreatedAt,
            IsTemplate = blueprint.Metadata.IsTemplate,
            TemplateCategory = blueprint.Metadata.TemplateCategory
        },
        Blocks = blueprint.Blocks
            .Select(b => new BlockYamlDto
            {
                Pos = new PositionDto { X = b.X, Y = b.Y, Z = b.Z },
                Id = b.BlockId,
                State = b.State?.Properties.ToDictionary(kvp => kvp.Key, kvp => kvp.Value),
                Tags = b.Tags.ToList()
            })
            .ToList()
    };

    if (blueprint.PhaseConfig is not null)
    {
        dto.Phases = new PhaseConfigDto
        {
            Mode = blueprint.PhaseConfig.Mode switch
            {
                PhaseMode.Override => "override",
                PhaseMode.Augment  => "augment",
                _                  => "auto"
            },
            List = blueprint.PhaseConfig.Phases
                .Select(p => new BuildPhaseDto
                {
                    Id = p.Id,
                    Description = p.Description,
                    Selectors = new PhaseSelectorsDto
                    {
                        Tags = p.Selectors.Tags.ToList(),
                        Ranges = p.Selectors.Ranges
                            .Select(r => new PhaseRangeDto { MinY = r.MinY, MaxY = r.MaxY })
                            .ToList()
                    }
                })
                .ToList()
        };
    }

    return dto;
}
```

### 4.1 Materials derivation

**Function**

```csharp
public static IReadOnlyList<MaterialSummary> DeriveMaterials(Blueprint blueprint)
{
    return blueprint.Blocks
        .GroupBy(b => b.BlockId)
        .Select(g => new MaterialSummary
        {
            BlockId = g.Key,
            Count = g.Count()
        })
        .OrderByDescending(m => m.Count)
        .ThenBy(m => m.BlockId)
        .ToArray();
}
```

### 4.2 Legend derivation

**Model**

```csharp
public sealed class LegendEntry
{
    public char Alias { get; init; }
    public string BlockId { get; init; } = default!;
}
```

**Function**

```csharp
public static IReadOnlyList<LegendEntry> DeriveLegend(IReadOnlyList<MaterialSummary> materials)
{
    var usedAliases = new HashSet<char>();
    var entries = new List<LegendEntry>();

    foreach (var material in materials)
    {
        var baseAlias = GetPreferredAlias(material.BlockId);
        var alias = baseAlias;

        while (usedAliases.Contains(alias))
        {
            alias = GetNextAlias(alias);
        }

        usedAliases.Add(alias);
        entries.Add(new LegendEntry { Alias = alias, BlockId = material.BlockId });
    }

    // Always reserve '.' for air
    if (!usedAliases.Contains('.'))
    {
        entries.Add(new LegendEntry { Alias = '.', BlockId = "air" });
        usedAliases.Add('.');
    }

    return entries;
}

private static char GetPreferredAlias(string blockId)
{
    // e.g. "minecraft:oak_planks" → 'P'
    var name = blockId.Split(':').Last();
    var firstLetter = name.FirstOrDefault(char.IsLetter);
    return firstLetter == default ? 'X' : char.ToUpperInvariant(firstLetter);
}

private static char GetNextAlias(char alias)
{
    // Simple fallback: cycle through A–Z, then digits
    if (alias >= 'A' && alias < 'Z') return (char)(alias + 1);
    if (alias == 'Z') return '0';
    if (alias >= '0' && alias < '9') return (char)(alias + 1);
    return 'X';
}
```

### 4.3 Layer grid derivation

**Model**

```csharp
public sealed class LayerGrid
{
    public int Y { get; init; }
    public string Name { get; init; } = default!;
    public string[] Rows { get; init; } = Array.Empty<string>(); // Z-major, each string length X
}
```

**Function**

```csharp
public static IReadOnlyList<LayerGrid> DeriveLayers(Blueprint blueprint, IReadOnlyList<LegendEntry> legend)
{
    var legendMap = legend.ToDictionary(l => l.BlockId, l => l.Alias);
    var xSize = blueprint.Dimensions.X;
    var ySize = blueprint.Dimensions.Y;
    var zSize = blueprint.Dimensions.Z;

    var layers = new List<LayerGrid>();

    for (var y = 0; y < ySize; y++)
    {
        var rows = new string[zSize];

        for (var z = 0; z < zSize; z++)
        {
            var chars = new char[xSize];

            for (var x = 0; x < xSize; x++)
            {
                var block = blueprint.Blocks.FirstOrDefault(b => b.X == x && b.Y == y && b.Z == z);
                if (block is null)
                {
                    chars[x] = '.'; // air
                }
                else
                {
                    if (!legendMap.TryGetValue(block.BlockId, out var alias))
                    {
                        alias = '?'; // unknown
                    }
                    chars[x] = alias;
                }
            }

            rows[z] = new string(chars);
        }

        layers.Add(new LayerGrid
        {
            Y = y,
            Name = $"Layer Y={y}",
            Rows = rows
        });
    }

    return layers;
}
```

### 4.4 Phase autogeneration

**Function**

```csharp
public static IReadOnlyList<BuildPhase> AutoGeneratePhases(Blueprint blueprint)
{
    // Simple heuristic MVP:
    // - foundation: Y == 0 or tags contain "foundation"
    // - walls: tags contain "wall" or 1 <= Y <= maxY - 2
    // - roof: tags contain "roof" or Y == maxY - 1
    // - interior: tags contain "interior"

    var maxY = blueprint.Dimensions.Y - 1;

    var foundationPhase = new BuildPhase
    {
        Id = "foundation",
        Description = "Floor and structural supports",
        Selectors = new PhaseSelectors
        {
            Tags = new[] { "foundation", "support" },
            Ranges = new[] { new PhaseRange { MinY = 0, MaxY = 0 } }
        }
    };

    var wallsPhase = new BuildPhase
    {
        Id = "walls",
        Description = "Walls and shell",
        Selectors = new PhaseSelectors
        {
            Tags = new[] { "wall" },
            Ranges = new[] { new PhaseRange { MinY = 1, MaxY = maxY - 1 } }
        }
    };

    var roofPhase = new BuildPhase
    {
        Id = "roof",
        Description = "Roof and top slabs",
        Selectors = new PhaseSelectors
        {
            Tags = new[] { "roof" },
            Ranges = new[] { new PhaseRange { MinY = maxY, MaxY = maxY } }
        }
    };

    var interiorPhase = new BuildPhase
    {
        Id = "interior",
        Description = "Furniture, lighting, and interior details",
        Selectors = new PhaseSelectors
        {
            Tags = new[] { "interior" }
        }
    };

    return new[] { foundationPhase, wallsPhase, roofPhase, interiorPhase };
}
```

### 4.5 Phase resolution (auto + overrides)

**Function**

```csharp
public static IReadOnlyList<BuildPhase> ResolvePhases(Blueprint blueprint)
{
    var auto = AutoGeneratePhases(blueprint);

    if (blueprint.PhaseConfig is null || blueprint.PhaseConfig.Mode == PhaseMode.Auto)
        return auto;

    if (blueprint.PhaseConfig.Mode == PhaseMode.Override)
        return blueprint.PhaseConfig.Phases;

    // Augment: merge auto + user-defined
    var user = blueprint.PhaseConfig.Phases;
    // Simple merge: user phases first, then auto phases that don't conflict by Id
    var userIds = new HashSet<string>(user.Select(p => p.Id));
    var merged = new List<BuildPhase>();
    merged.AddRange(user);
    merged.AddRange(auto.Where(p => !userIds.Contains(p.Id)));
    return merged;
}
```

### 4.6 Markdown documentation generation

**Function outline**

```csharp
public static string GenerateMarkdown(Blueprint blueprint)
{
    var materials = DeriveMaterials(blueprint);
    var legend = DeriveLegend(materials);
    var layers = DeriveLayers(blueprint, legend);
    var phases = ResolvePhases(blueprint);

    var sb = new StringBuilder();

    sb.AppendLine($"# {blueprint.Name}");
    sb.AppendLine();
    sb.AppendLine($"**Id:** `{blueprint.Id}`");
    sb.AppendLine($"**Dimensions:** {blueprint.Dimensions.X} × {blueprint.Dimensions.Z} × {blueprint.Dimensions.Y}");
    sb.AppendLine();

    sb.AppendLine("## Materials");
    sb.AppendLine();
    sb.AppendLine("| Block | Quantity |");
    sb.AppendLine("|---|---|");
    foreach (var m in materials)
    {
        sb.AppendLine($"| `{m.BlockId}` | {m.Count} |");
    }
    sb.AppendLine();

    sb.AppendLine("## Layers");
    sb.AppendLine();
    foreach (var layer in layers)
    {
        sb.AppendLine($"### Y={layer.Y}");
        foreach (var row in layer.Rows)
        {
            sb.AppendLine(row);
        }
        sb.AppendLine();
    }

    sb.AppendLine("## Legend");
    sb.AppendLine();
    sb.AppendLine("| Alias | Block |");
    sb.AppendLine("|---|---|");
    foreach (var entry in legend)
    {
        sb.AppendLine($"| `{entry.Alias}` | `{entry.BlockId}` |");
    }
    sb.AppendLine();

    sb.AppendLine("## Phases");
    sb.AppendLine();
    foreach (var phase in phases)
    {
        sb.AppendLine($"- **{phase.Id}:** {phase.Description}");
    }

    return sb.ToString();
}
```

## 5. Blazor editor — components and data flow

### 5.1 State container

Use a scoped service to hold the current blueprint and derived artifacts.

```csharp
public sealed class BlueprintEditorState
{
    public Blueprint? CurrentBlueprint { get; private set; }

    public IReadOnlyList<MaterialSummary> Materials { get; private set; } = Array.Empty<MaterialSummary>();
    public IReadOnlyList<LegendEntry> Legend { get; private set; } = Array.Empty<LegendEntry>();
    public IReadOnlyList<LayerGrid> Layers { get; private set; } = Array.Empty<LayerGrid>();
    public IReadOnlyList<BuildPhase> Phases { get; private set; } = Array.Empty<BuildPhase>();

    public event Action? StateChanged;

    public void Load(Blueprint blueprint)
    {
        CurrentBlueprint = blueprint;
        RecomputeDerived();
        StateChanged?.Invoke();
    }

    public void UpdateBlueprint(Blueprint blueprint)
    {
        CurrentBlueprint = blueprint;
        RecomputeDerived();
        StateChanged?.Invoke();
    }

    private void RecomputeDerived()
    {
        if (CurrentBlueprint is null)
        {
            Materials = Array.Empty<MaterialSummary>();
            Legend = Array.Empty<LegendEntry>();
            Layers = Array.Empty<LayerGrid>();
            Phases = Array.Empty<BuildPhase>();
            return;
        }

        Materials = DeriveMaterials(CurrentBlueprint);
        Legend = DeriveLegend(Materials);
        Layers = DeriveLayers(CurrentBlueprint, Legend);
        Phases = ResolvePhases(CurrentBlueprint);
    }
}
```

```csharp
private async Task HandleBlockChanged(PlacementBlock? newBlock)
{
    var blueprint = _state.CurrentBlueprint!;
    var blocks = blueprint.Blocks.ToList();

    // Remove existing block at position
    blocks.RemoveAll(b => b.X == _selectedBlockPosition.X &&
                          b.Y == _selectedBlockPosition.Y &&
                          b.Z == _selectedBlockPosition.Z);

    if (newBlock is not null)
    {
        blocks.Add(newBlock);
    }

    var updated = blueprint with { Blocks = blocks };
    _state.UpdateBlueprint(updated);
}
```

### 7.3 HTN planner integration

In `HtnTaskLibrary.DecomposeBuild()`:

- Use `BlueprintParser` to get `Blueprint`.
- Use `ResolvePhases(blueprint)` to get phases.
- For each phase:
  - Filter blocks by tags/Y ranges.
  - Emit `PlaceBlock` tasks accordingly.

Example:

```csharp
public IEnumerable<Task> DecomposeBuild(string blueprintId)
{
    var blueprint = _blueprintParser.Parse(blueprintId);
    var phases = ResolvePhases(blueprint);

    foreach (var phase in phases)
    {
        var phaseBlocks = SelectBlocksForPhase(blueprint, phase);
        foreach (var block in phaseBlocks)
        {
            yield return new PlaceBlockTask(block);
        }
    }
}

private static IReadOnlyList<PlacementBlock> SelectBlocksForPhase(Blueprint blueprint, BuildPhase phase)
{
    var blocks = blueprint.Blocks.AsEnumerable();

    if (phase.Selectors.Tags.Count > 0)
    {
        blocks = blocks.Where(b => b.Tags.Any(t => phase.Selectors.Tags.Contains(t)));
    }

    foreach (var range in phase.Selectors.Ranges)
    {
        blocks = blocks.Where(b =>
            (!range.MinY.HasValue || b.Y >= range.MinY.Value) &&
            (!range.MaxY.HasValue || b.Y <= range.MaxY.Value));
    }

    return blocks.ToArray();
}
```
